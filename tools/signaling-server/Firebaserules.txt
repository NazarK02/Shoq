rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function isSignedIn() {
      return request.auth != null;
    }

    function conversationExists(conversationId) {
      return exists(
        /databases/$(database)/documents/conversations/$(conversationId)
      );
    }

    function conversationDoc(conversationId) {
      return get(
        /databases/$(database)/documents/conversations/$(conversationId)
      );
    }

    function isConversationParticipant(conversationId) {
      let conversation = conversationDoc(conversationId);
      return isSignedIn() &&
        conversationExists(conversationId) &&
        conversation.data.participants is list &&
        conversation.data.participants.hasAny([request.auth.uid]);
    }

    function isServerManager(conversationId) {
      let conversation = conversationDoc(conversationId);
      let admins = conversation.data.admins is list ? conversation.data.admins : [];
      return isSignedIn() &&
        conversationExists(conversationId) &&
        conversation.data.type == 'server' &&
        (
          conversation.data.createdBy == request.auth.uid ||
          request.auth.uid in admins
        );
    }

    // Users collection
    match /users/{userId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && request.auth.uid == userId;
    }

    // Friend requests collection
    match /friendRequests/{friendRequestId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null;
      allow update, delete: if request.auth != null;
    }

    // Contacts collection (friends)
    match /contacts/{userId}/friends/{friendId} {
      allow read: if request.auth != null && request.auth.uid == userId;
      allow write: if request.auth != null && request.auth.uid == userId;
      allow create: if request.auth != null && request.auth.uid == friendId;
      allow delete: if request.auth != null &&
                       (request.auth.uid == userId || request.auth.uid == friendId);
    }

    // Blocked users collection
    match /contacts/{userId}/blocked/{blockedId} {
      allow read: if request.auth != null && request.auth.uid == userId;
      allow write: if request.auth != null && request.auth.uid == userId;
    }

    // Notifications
    match /notifications/{notificationId} {
      allow create: if request.auth != null;
      allow read, update: if request.auth != null &&
                         resource.data.recipientId == request.auth.uid;
    }

    // Notifications settings
    match /users/{userId}/settings/{document} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }

    // Server invite links
    match /serverInvites/{inviteCode} {
      function hasServerId(data) {
        return data.serverId is string && data.serverId.size() > 0;
      }

      function isJoinCounterUpdateOnly() {
        let diff = request.resource.data.diff(resource.data);
        let beforeCount = resource.data.useCount is number ? resource.data.useCount : 0;
        return diff.affectedKeys().hasOnly(['useCount', 'lastAcceptedAt']) &&
          request.resource.data.useCount is number &&
          request.resource.data.useCount >= beforeCount;
      }

      allow read: if isSignedIn();

      allow create: if isSignedIn() &&
        request.resource.data.code == inviteCode &&
        request.resource.data.createdBy == request.auth.uid &&
        hasServerId(request.resource.data) &&
        isServerManager(request.resource.data.serverId);

      allow update: if isSignedIn() &&
        hasServerId(resource.data) &&
        (
          isServerManager(resource.data.serverId) ||
          (
            resource.data.revokedAt == null &&
            isConversationParticipant(resource.data.serverId) &&
            isJoinCounterUpdateOnly()
          )
        );

      allow delete: if isSignedIn() &&
        hasServerId(resource.data) &&
        isServerManager(resource.data.serverId);
    }

    // Conversations
    match /conversations/{conversationId} {
      function participantsList(data) {
        return data.participants is list ? data.participants : [];
      }

      function inviteCodeForJoin() {
        let requestCode = request.resource.data.activeInviteCode is string
          ? request.resource.data.activeInviteCode
          : '';
        return requestCode.size() > 0
          ? requestCode
          : (
              resource.data.activeInviteCode is string
                ? resource.data.activeInviteCode
                : ''
            );
      }

      function activeInviteRef(inviteCode) {
        return /databases/$(database)/documents/serverInvites/$(inviteCode);
      }

      function canJoinServerViaInvite() {
        let beforeParticipants = participantsList(resource.data);
        let afterParticipants = participantsList(request.resource.data);
        let inviteCode = inviteCodeForJoin();
        return request.auth != null &&
          resource.data.type == 'server' &&
          !(request.auth.uid in beforeParticipants) &&
          request.auth.uid in afterParticipants &&
          afterParticipants.hasAll(beforeParticipants) &&
          afterParticipants.size() <= beforeParticipants.size() + 1 &&
          request.resource.data.diff(resource.data).changedKeys().hasOnly([
            'participants',
            'updatedAt',
            'activeInviteCode'
          ]) &&
          inviteCode.size() > 0 &&
          exists(activeInviteRef(inviteCode)) &&
          get(activeInviteRef(inviteCode)).data.serverId == conversationId &&
          get(activeInviteRef(inviteCode)).data.revokedAt == null;
      }

      allow create: if request.auth != null &&
        request.resource.data.participants is list &&
        request.auth.uid in request.resource.data.participants;

      allow read: if request.auth != null &&
        request.auth.uid in resource.data.participants;

      allow update: if request.auth != null &&
        (
          request.auth.uid in resource.data.participants ||
          canJoinServerViaInvite()
        );

      match /voiceChannels/{channelId} {
        function isConversationParticipant() {
          return request.auth != null &&
            request.auth.uid in get(/databases/$(database)/documents/conversations/$(conversationId)).data.participants;
        }

        allow read: if isConversationParticipant();
        allow create, update: if isConversationParticipant();

        match /messages/{messageId} {
          allow read: if isConversationParticipant();

          allow create: if isConversationParticipant() &&
            request.auth.uid == request.resource.data.senderId &&
            request.resource.data.text is string &&
            request.resource.data.text.size() > 0 &&
            request.resource.data.text.size() <= 1500;

          allow update, delete: if isConversationParticipant() &&
            request.auth.uid == resource.data.senderId;
        }

        allow delete: if false;
      }

      match /messages/{messageId} {
        function isConversationParticipant() {
          return request.auth != null &&
            request.auth.uid in get(/databases/$(database)/documents/conversations/$(conversationId)).data.participants;
        }

        function reactionsOrEmpty(data) {
          return data.reactions is map ? data.reactions : {};
        }

        function isValidReactionValue(value) {
          return value is string ||
            (value is list &&
             value.size() > 0 &&
             value.size() <= 12 &&
             value[0] is string);
        }

        function isReadReceiptUpdate() {
          return request.resource.data.diff(resource.data).changedKeys().hasOnly(['read', 'readAt']) &&
            request.resource.data.read == true;
        }

        function isOwnReactionUpdate() {
          let before = reactionsOrEmpty(resource.data);
          let after = reactionsOrEmpty(request.resource.data);
          let reactionDiff = after.diff(before);
          return request.resource.data.diff(resource.data).changedKeys().hasOnly(['reactions']) &&
            reactionDiff.addedKeys().hasOnly([request.auth.uid]) &&
            reactionDiff.changedKeys().hasOnly([request.auth.uid]) &&
            reactionDiff.removedKeys().hasOnly([request.auth.uid]) &&
            (
              !(request.auth.uid in after) ||
              isValidReactionValue(after[request.auth.uid])
            );
        }

        function isCallOfferCreate() {
          return request.resource.data.type == 'call_offer' &&
            request.resource.data.encrypted == false &&
            request.resource.data.read == true &&
            request.resource.data.status == 'ringing' &&
            request.resource.data.callId is string &&
            request.resource.data.callId.size() > 0 &&
            request.resource.data.toId is string &&
            request.resource.data.toId.size() > 0 &&
            request.resource.data.toId in get(/databases/$(database)/documents/conversations/$(conversationId)).data.participants &&
            request.resource.data.senderId == request.auth.uid &&
            request.resource.data.offer is map &&
            request.resource.data.offer.sdp is string &&
            request.resource.data.offer.sdp.size() > 0 &&
            request.resource.data.offer.type is string &&
            request.resource.data.offer.type.size() > 0 &&
            request.auth.uid in get(/databases/$(database)/documents/conversations/$(conversationId)).data.participants &&
            messageId == ('call_' + request.resource.data.callId);
        }

        function isCallOfferStatusUpdate() {
          let newStatus = request.resource.data.status;
          return request.resource.data.type == 'call_offer' &&
            request.auth.uid in get(/databases/$(database)/documents/conversations/$(conversationId)).data.participants &&
            (
              request.auth.uid == resource.data.senderId ||
              request.auth.uid == resource.data.toId
            ) &&
            request.resource.data.callId == resource.data.callId &&
            newStatus in ['answered', 'ended', 'declined', 'missed'] &&
            request.resource.data.diff(resource.data).changedKeys().hasOnly([
              'status',
              'updatedAt',
              'answeredAt',
              'endedAt',
              'declinedAt',
              'missedAt'
            ]);
        }

        allow read: if request.auth != null &&
          request.auth.uid in get(/databases/$(database)/documents/conversations/$(conversationId)).data.participants;

        allow create: if request.auth != null &&
          request.auth.uid in get(/databases/$(database)/documents/conversations/$(conversationId)).data.participants &&
          request.auth.uid == request.resource.data.senderId &&
          (
            // Encrypted message
            ((request.resource.data.type in [
                'text',
                'sticker',
                'location',
                'location_live',
                'file'
              ]) &&
             request.resource.data.encrypted == true &&
             (
               request.resource.data.ciphertext is string ||
               request.resource.data.ciphertexts is map
             ) &&
             (
               !('senderCiphertext' in request.resource.data) ||
               request.resource.data.senderCiphertext is string
             ) &&
             (
               !('senderCiphertexts' in request.resource.data) ||
               request.resource.data.senderCiphertexts is map
             ) &&
             (
               !('senderPublicKey' in request.resource.data) ||
               request.resource.data.senderPublicKey is string
             ))

            ||

              // File message
              (request.resource.data.type == 'file' &&
               request.resource.data.encrypted == false &&
               request.resource.data.fileUrl is string &&
               request.resource.data.fileName is string &&
               (request.resource.data.fileSize is int || request.resource.data.fileSize is number) &&
               request.resource.data.mimeType is string &&
               request.resource.data.storagePath is string)

              ||

              // Call offer / call state message
              isCallOfferCreate()
            );

        allow update: if isConversationParticipant() &&
          (
            request.auth.uid == resource.data.senderId ||
            isReadReceiptUpdate() ||
            isOwnReactionUpdate() ||
            isCallOfferStatusUpdate()
          );

        allow delete: if request.auth != null &&
          request.auth.uid == resource.data.senderId;
      }
    }
  }
}
